SELECT 
    ProductID,
    productName,
    SUM(UnitsInStock + UnitsOnOrder) AS totalQuantity,
    CONCAT('$',
            (UnitsInStock + UnitsOnOrder) * UnitPrice) AS totalPrice,
    CONCAT('$', AVG(QuantityPerUnit * UnitPrice)) AS WeighedAvgPrice,
    CONCAT('$', AVG(QuantityPerUnit)) AS AvgPrice,
    CONCAT('$', UnitPrice) AS CurrentPrice
FROM
    products
WHERE
    UnitsInStock IS NOT NULL
GROUP BY ProductID

select p.ProductID, P.ProductName,P.SupplierID,s.CompanyName, 
case when s.Phone is NULL then ' ' 
else s.Phone end as 'Phone' ,
MIN(100-p.UnitsInStock+p.UnitsOnOrder) as 'to order'
from products as p join suppliers as s
on p.ProductID = s.SupplierID
where  p.UnitsInStock+p.UnitsOnOrder < 10
group by p.ProductID

SELECT 
    p.ProductID,
    P.ProductName,
    P.SupplierID,
    s.CompanyName,
    CASE
        WHEN s.Phone IS NULL THEN ' '
        ELSE s.Phone
    END AS 'Phone',
    MIN(100 - p.UnitsInStock + p.UnitsOnOrder) AS 'to order'
FROM
    products AS p
        JOIN
    suppliers AS s ON p.ProductID = s.SupplierID
WHERE
    p.UnitsInStock + p.UnitsOnOrder < 10
GROUP BY p.ProductID

SELECT 
    p.ProductName,
    s.CompanyName,
    YEAR(o.OrderDate) AS OrderYear,
    o.OrderDate,
    QUARTER(o.OrderDate) AS testing,
    IF(QUARTER(o.OrderDate) = 1,
        od.UnitPrice * od.Quantity * (1 - od.Discount),
        0) AS Qrt1,
    IF(QUARTER(o.OrderDate) = 2,
        od.UnitPrice * od.Quantity * (1 - od.Discount),
        0) AS Qrt2,
    IF(QUARTER(o.OrderDate) = 3,
        od.UnitPrice * od.Quantity * (1 - od.Discount),
        0) AS Qrt3,
    IF(QUARTER(o.OrderDate) = 4,
        od.UnitPrice * od.Quantity * (1 - od.Discount),
        0) AS Qrt4
FROM
    orders AS o
        JOIN
    `order details` AS od ON o.OrderID = od.OrderID
        JOIN
    products AS p ON p.ProductID = od.ProductID
        JOIN
    suppliers AS s ON p.SupplierID = s.SupplierID
ORDER BY p.ProductName , s.CompanyName;
